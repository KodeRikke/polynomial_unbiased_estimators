\section*{Quadratic Function Report}
\section*{Polynomial}
\begin{dmath*}
f(q) = a q^{2} + b q + c
\end{dmath*}
\section*{Naive}
\begin{dmath*}
g_{\mathrm{naive}}(X) = X^{2} a + X b + c
\end{dmath*}
\begin{dmath*}
\mathbb{E}[g_{\mathrm{naive}}(X)] = 2 a {\frac{\Delta}{\epsilon}}^{2} + a q^{2} + b q + c
\end{dmath*}
\begin{dmath*}
\operatorname{Var}(g_{\mathrm{naive}}(X)) = 2 {\frac{\Delta}{\epsilon}}^{2} \left(10 a^{2} {\frac{\Delta}{\epsilon}}^{2} + 4 a^{2} q^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\begin{dmath*}
\operatorname{MSE}(g_{\mathrm{naive}}) = 2 {\frac{\Delta}{\epsilon}}^{2} \left(12 a^{2} {\frac{\Delta}{\epsilon}}^{2} + 4 a^{2} q^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\begin{dmath*}
\operatorname{Bias}(g_{\mathrm{naive}}) = 2 a {\frac{\Delta}{\epsilon}}^{2}
\end{dmath*}
\section*{Unbiased}
\begin{dmath*}
g_{\mathrm{unb}}(X) = X^{2} a + X b - 2 a {\frac{\Delta}{\epsilon}}^{2} + c
\end{dmath*}
\begin{dmath*}
\mathbb{E}[g_{\mathrm{unb}}(X)] = a q^{2} + b q + c
\end{dmath*}
\begin{dmath*}
\operatorname{Var}(g_{\mathrm{unb}}(X)) = 2 {\frac{\Delta}{\epsilon}}^{2} \left(10 a^{2} {\frac{\Delta}{\epsilon}}^{2} + 4 a^{2} q^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\begin{dmath*}
\operatorname{MSE}(g_{\mathrm{unb}}) = 2 {\frac{\Delta}{\epsilon}}^{2} \left(10 a^{2} {\frac{\Delta}{\epsilon}}^{2} + 4 a^{2} q^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\section*{Comparison}
\begin{dmath*}
\text{mean gap} = - 2 a {\frac{\Delta}{\epsilon}}^{2}
\end{dmath*}
\begin{dmath*}
\text{variance gap} = 0
\end{dmath*}
\begin{dmath*}
\text{MSE gap} = - 4 a^{2} {\frac{\Delta}{\epsilon}}^{4}
\end{dmath*}
\begin{dmath*}
\operatorname{Bias}(g_{\mathrm{naive}})^2 = 4 a^{2} {\frac{\Delta}{\epsilon}}^{4}
\end{dmath*}
