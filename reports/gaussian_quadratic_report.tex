\section*{Gaussian Quadratic Function Report}
\section*{Polynomial}
\begin{dmath*}
f(q) = a q^{2} + b q + c
\end{dmath*}
\section*{Naive}
\begin{dmath*}
g_{\mathrm{naive}}(X) = Q^{2} a + Q b + c
\end{dmath*}
\begin{dmath*}
\mathbb{E}[g_{\mathrm{naive}}(X)] = a q^{2} + a \sigma^{2} + b q + c
\end{dmath*}
\begin{dmath*}
\operatorname{Var}(g_{\mathrm{naive}}(X)) = \sigma^{2} \left(4 a^{2} q^{2} + 2 a^{2} \sigma^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\begin{dmath*}
\operatorname{MSE}(g_{\mathrm{naive}}) = \sigma^{2} \left(4 a^{2} q^{2} + 3 a^{2} \sigma^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\begin{dmath*}
\operatorname{Bias}(g_{\mathrm{naive}}) = a \sigma^{2}
\end{dmath*}
\section*{Unbiased}
\begin{dmath*}
g_{\mathrm{unb}}(X) = Q^{2} a + Q b - a \sigma^{2} + c
\end{dmath*}
\begin{dmath*}
\mathbb{E}[g_{\mathrm{unb}}(X)] = a q^{2} + b q + c
\end{dmath*}
\begin{dmath*}
\operatorname{Var}(g_{\mathrm{unb}}(X)) = \sigma^{2} \left(4 a^{2} q^{2} + 2 a^{2} \sigma^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\begin{dmath*}
\operatorname{MSE}(g_{\mathrm{unb}}) = \sigma^{2} \left(4 a^{2} q^{2} + 2 a^{2} \sigma^{2} + 4 a b q + b^{2}\right)
\end{dmath*}
\section*{Comparison}
\begin{dmath*}
\text{mean gap} = - a \sigma^{2}
\end{dmath*}
\begin{dmath*}
\text{variance gap} = 0
\end{dmath*}
\begin{dmath*}
\text{MSE gap} = - a^{2} \sigma^{4}
\end{dmath*}
\begin{dmath*}
\operatorname{Bias}(g_{\mathrm{naive}})^2 = a^{2} \sigma^{4}
\end{dmath*}
